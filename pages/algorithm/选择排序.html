  1. 这样第一位放了最小值, 那让上面代码, 多执行几次是不是就好了呢
let arr = [4, 2, 3, 6, 5];
for (let i = 0; i < arr.length; i++) {
    // 注意这里不能再用0了, 因为第二次执行, 0位置已经是最小值了, 所以这里直接用i, 让索引继续向后
    let minIndex = i;
    // 这里也注意不要用0, 因为0位置放了最小值, 而且也不要用i, 自己和自己比没有意义
    for (let j = i + 1; j < arr.length; j++) {
        if (arr[j] < arr[minIndex]) {
            minIndex = j;
        }
    }
    let c = arr[i];
    arr[i] = arr[minIndex];
    arr[minIndex] = c;
}
console.log(arr);

# 优化1

当外层循环到最后一次, 没有必要再比较了, 所以最外层可以-1 

let arr = [4, 2, 3, 6, 5];
for (let i = 0; i < arr.length - 1; i++) {
    let minIndex = i;
    for (let j = i + 1; j < arr.length; j++) {
        if (arr[j] < arr[minIndex]) {
            minIndex = j;
        }
    }
    let c = arr[i];
    arr[i] = arr[minIndex];
    arr[minIndex] = c;
}
console.log(arr);

# 优化2

内层for循环后, 发现i就是最小值的下标, 不需要交换, 所以给判断加个判断条件 

let arr = [4, 2, 3, 6, 5];
for (let i = 0; i < arr.length - 1; i++) {
    let minIndex = i;
    for (let j = i + 1; j < arr.length; j++) {
        if (arr[j] < arr[minIndex]) {
            minIndex = j;
        }
    }

    if (minIndex !== i){
        let c = arr[i];
        arr[i] = arr[minIndex];
        arr[minIndex] = c;
    }
}
console.log(arr);
