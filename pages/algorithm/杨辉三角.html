# 取第m行, 第n个位置的值

思路看图:

每次调用方法, 求 [ i - 1 ] [ j - 1 ] + [ i - 1 ][ j ] 的结果, 当遇到m === 1或者 n === m, 直接返回1即可 (为什么是1, 因为这次从1开始数, 而非数组下标0)

function fn(n, m) { // n行数, 第m个
    if (m === 1 || n === m) { // m为开头的肯定是1, n和m相等为最后一个元素为1
        return 1;
    } else {
        return fn(n - 1, m - 1) + fn(n - 1, m);
    }
}
console.log(fn(7, 5)); // 15
